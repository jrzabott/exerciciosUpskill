	/* lim, num, menorN, soma INTEIRO */
	lim, num, menorN, soma REAL
ALG
INICIO
    lim <- 0
    soma <- 0
    REPETIR
        LER lim
        SE (lim < 0)
            ESCREVER ("LIMITE DEVE SER POSITIVO")
        FIM SE
    ENQUANTO (lim < 0)
    REPETIR
        LER num
        SE (num < 0)
            ESCREVER ("NUMERO DEVE SER POSITIVO")
        FIM SE
    ENQUANTO (num < 0)
    menorN <- num   /* menorN inicial Ã© o primeiro numero inserido */
    ENQUANTO (soma < lim)
        soma <- soma + num
        REPETE
            LER num
            SE (num < 0)
                ESCREVER ("NUMERO DEVE SER POSITIVO.")
            FIM SE
        ENQUANTO (num < 0)
        SE (num < menorN)
            menorN <- num
        FIM SE
    FIM ENQUANTO
    ESCREVER (menorN)
FIM
